//custom... drake card normally removes size penalty on weapon, but since I don't have size penalty, I had to get creative
//takes 12% of each of 5 main stats and puts half those points into the 6th, based on what slot the card is in (it can go in any).
Item("Drake_Card") {
OnEquip: 
	var str = GetStat(Str) * 12 / 100;
	var agi = GetStat(Agi) * 12 / 100;
	var mnd = GetStat(Int) * 12 / 100; //int is a reserved keyword in c#
	var dex = GetStat(Dex) * 12 / 100;
	var luk = GetStat(Luk) * 12 / 100;
	var vit = GetStat(Vit) * 12 / 100;
	
	if(IsInPosition(Headgear) == false) 	AddStat(AddInt, mnd * -1); else AddStat(AddInt, (str + agi + dex + luk + vit) / 2);
	if(IsInPosition(Body) == false) 		AddStat(AddVit, vit * -1); else AddStat(AddVit, (str + agi + dex + luk + mnd) / 2);
	if(IsInPosition(AnyHand) == false)	 	AddStat(AddStr, str * -1); else AddStat(AddStr, (mnd + agi + dex + luk + vit) / 2);
	if(IsInPosition(Garment) == false) 		AddStat(AddAgi, agi * -1); else AddStat(AddAgi, (str + mnd + dex + luk + vit) / 2);
	if(IsInPosition(Footgear) == false) 	AddStat(AddDex, dex * -1); else AddStat(AddDex, (str + agi + mnd + luk + vit) / 2);
	if(IsInPosition(Accessory) == false) 	AddStat(AddLuk, luk * -1); else AddStat(AddLuk, (str + agi + dex + mnd + vit) / 2);
}
